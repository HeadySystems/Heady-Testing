Python Bundler

Run: python3 bundle.py

#!/usr/bin/env python3
import re, os, zipfile, shutil

MD='05-mcp-server-tools-transport.md'
OUT='heady-mcp-extracted'
ZIP='heady-mcp-complete.zip'

print('📖 Reading specification...')
try:
    content = open(MD, encoding='utf-8').read()
except FileNotFoundError:
    print(f'❌ {MD} not found! Make sure it\'s in this directory.')
    exit(1)

pattern = r'###\s*`([^`]+)`\s*\n+```[\w-]*\n(.*?)```'
matches = re.findall(pattern, content, re.DOTALL)
print(f'✅ Found {len(matches)} source files\n')

if os.path.exists(OUT): shutil.rmtree(OUT)
os.makedirs(OUT, exist_ok=True)

for filepath, code in matches:
    fp = filepath.strip()
    full = os.path.join(OUT, fp)
    os.makedirs(os.path.dirname(full), exist_ok=True)
    open(full, 'w', encoding='utf-8').write(code)
    print(f'  ✅ {fp}')

docs = os.path.join(OUT, 'docs')
os.makedirs(docs, exist_ok=True)
print('\n📚 Copying documentation...')
for doc in ['Heady_Service_Reference.docx', 'Heady_System_Architecture_Overview.docx', 'Heady_Development_Deployment_Guide.docx']:
    if os.path.exists(doc):
        shutil.copy2(doc, docs)
        print(f'  ✅ {doc}')

shutil.copy2(MD, OUT)
print(f'  ✅ {MD}')

with open(os.path.join(OUT, 'MANIFEST.txt'), 'w') as f:
    f.write(f'Heady MCP Complete Bundle\n{"="*50}\n')
    f.write(f'Source files: {len(matches)}\nDocs: 4 files\n')

print(f'\n📦 Creating {ZIP}...')
with zipfile.ZipFile(ZIP, 'w', zipfile.ZIP_DEFLATED) as z:
    for root, _, files in os.walk(OUT):
        for file in files:
            fp = os.path.join(root, file)
            z.write(fp, os.path.relpath(fp, OUT))

size = os.path.getsize(ZIP) / 1024
print(f'✅ Created {ZIP} ({size:.1f} KB)\n')
print('='*50)
print('✅ COMPLETE!')
print(f'📦 {ZIP}')
print('='*50)

Node.js Bundler

Run: node bundle.js

#!/usr/bin/env node
const fs=require('fs'),path=require('path'),{execSync}=require('child_process');
const MD='05-mcp-server-tools-transport.md',OUT='heady-mcp-extracted',ZIP='heady-mcp-complete.zip';

console.log('📖 Reading specification...');
let content;
try { content=fs.readFileSync(MD,'utf8'); }
catch { console.log(`❌ ${MD} not found!`); process.exit(1); }

const re=/###\s+`([^`]+)`\s*\n+```[\w-]*\n([\s\S]*?)```/g;
const files=[]; let m;
while((m=re.exec(content))) files.push({path:m[1].trim(),code:m[2]});
console.log(`✅ Found ${files.length} source files\n`);

if(fs.existsSync(OUT))fs.rmSync(OUT,{recursive:true});
fs.mkdirSync(OUT,{recursive:true});

for(const f of files){
  const full=path.join(OUT,f.path);
  fs.mkdirSync(path.dirname(full),{recursive:true});
  fs.writeFileSync(full,f.code);
  console.log(`  ✅ ${f.path}`);
}

const docs=path.join(OUT,'docs');
fs.mkdirSync(docs,{recursive:true});
console.log('\n📚 Copying documentation...');
for(const d of ['Heady_Service_Reference.docx','Heady_System_Architecture_Overview.docx','Heady_Development_Deployment_Guide.docx']){
  if(fs.existsSync(d)){ fs.copyFileSync(d,path.join(docs,d)); console.log(`  ✅ ${d}`); }
}
fs.copyFileSync(MD,path.join(OUT,MD));
console.log(`  ✅ ${MD}`);

fs.writeFileSync(path.join(OUT,'MANIFEST.txt'),
  `Heady MCP Complete Bundle\n${'='.repeat(50)}\nSource files: ${files.length}\nDocs: 4 files\n`);

console.log(`\n📦 Creating ${ZIP}...`);
try{
  execSync(`cd "${OUT}"&&zip -r "../${ZIP}" . -x "*.DS_Store"`,{stdio:'pipe'});
  const sz=(fs.statSync(ZIP).size/1024).toFixed(1);
  console.log(`✅ Created ${ZIP} (${sz} KB)\n`);
}catch{
  console.log(`⚠️  zip command not found. Files extracted to ./${OUT}/`);
}
console.log('='.repeat(50));
console.log('✅ COMPLETE!');
console.log(`📦 ${ZIP}`);
console.log('='.repeat(50));

Bash Bundler

Run: bash bundle.sh

#!/bin/bash
echo "🎯 Heady MCP Quick Bundle"
echo "Bundles 4 documentation files (no extraction)"
echo ""

if [ ! -f "05-mcp-server-tools-transport.md" ]; then
    echo "❌ 05-mcp-server-tools-transport.md not found!"
    exit 1
fi

ZIP="heady-mcp-docs-$(date +%Y%m%d).zip"

echo "📦 Creating $ZIP..."
zip -q "$ZIP" \
    05-mcp-server-tools-transport.md \
    Heady_Service_Reference.docx \
    Heady_System_Architecture_Overview.docx \
    Heady_Development_Deployment_Guide.docx 2>/dev/null

if [ -f "$ZIP" ]; then
    SIZE=$(ls -lh "$ZIP" | awk '{print $5}')
    echo "✅ Created $ZIP ($SIZE)"
    echo ""
    echo "Note: This bundles the 4 files. For source extraction, use Python or Node.js bundler."
else
    echo "❌ Failed to create ZIP"
fi

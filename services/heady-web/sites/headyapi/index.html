Code Examples

Get started in minutes with our SDKs

curl -X POST https://api.heady.io/v1/query \ -H "Content-Type: application/json" \ -H "Authorization: Bearer sk_live_abc123" \ -d '{ "query": "{ user(id: \"usr_123\") { name email } }" }'
from heady import Client client = Client(api_key="sk_live_abc123") response = client.query(""" query GetUser($id: ID!) { user(id: $id) { name email } } """, variables={"id": "usr_123"}) print(response.data)
import { HeadyClient } from '@heady/sdk'; const client = new HeadyClient({ apiKey: 'sk_live_abc123' }); const { data } = await client.query(` query GetUser($id: ID!) { user(id: $id) { name email } } `, { id: 'usr_123' }); console.log(data);
